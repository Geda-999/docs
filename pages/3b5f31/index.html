//utils/eventBus.js
class EventBus {
  constructor() {
    this.list = [];
  }
  //发布
  $emit(eventName, options) {
    setTimeout(() => {
      this.list.forEach((item) => {
        if (item.eventName === eventName) {
          item.callBack(options);
        }
      });
    }, 0);
  }
  //订阅
  $on(eventName, callBack) {
    this.list.push({ eventName, callBack });
    console.log(this.list);
  }
}
export default new EventBus();

# 9、微信登陆

		// 登入
		wx.login({
